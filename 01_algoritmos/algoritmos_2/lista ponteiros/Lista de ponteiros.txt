#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

void troca(int *ptr1, int *ptr2);

int main()
{
    setlocale(LC_ALL, "portuguese");
    int n1, n2;

    printf("Digite um n·mero: ");
    scanf("%d%*c", &n1);
    printf("Digite outro n·mero: ");
    scanf("%d%*c", &n2);
    troca(&n1, &n2);
    printf("\nn1: %d | n2: %d", n1, n2);

    return 0;
}

void troca(int *ptr1, int *ptr2)
{
    *ptr1 += *ptr2;
    *ptr2 = *ptr1 - *ptr2;
    *ptr1 -= *ptr2;
}

    printf("\nn1: %d | n2: %d", n1, n2);

void maior_valor(int *ptr1, int *ptr2)
{
    if (*ptr2 > *ptr1)
    {
        *ptr1 += *ptr2;
        *ptr2 = *ptr1 - *ptr2;
        *ptr1 -= *ptr2;
    }
}

    int a, b, soma_dobro;

    printf("Digite outro n·mero: ");
    scanf("%d%*c", &b);

    soma_dobro = soma(&a, &b);

    printf("\nA*2: [%d],\nB*2: [%d],\nSoma de A e B: [%d].", a, b, soma_dobro);

int soma(int *ptr1, int *ptr2)
{
    *ptr1 *= 2;
    *ptr2 *= 2;

    return *ptr1 + *ptr2;
}
